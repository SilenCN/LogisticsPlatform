create database LogisticsPlatform;


create table user(
    id INTEGER primary key AUTO_INCREMENT,
    name TEXT NOT NULL,
    password TEXT NOT NULL,
    type INTEGER NOT NULL,
    info BOOLEAN,
    createTime BIGINT);

create table carInfo(
    id INTEGER primary key,
    phone TEXT,
    carType INTEGER,
    carNumber TEXT,
    transportType TEXT
    );

create table ownerInfo(
    id INTEGER primary key,
    phone TEXT,
    address TEXT,
    goods INTEGER
    );



create table orders(
    id INTEGER primary key AUTO_INCREMENT,
    ownerId INTEGER NOT NULL,
    status INTEGER,
    createTime BIGINT);

create table orderInfo(
    id INTEGER primary key,
    type INTEGER,
    goods TEXT,
    weight FLOAT,
    departure TEXT,
    target TEXT,
    remark TEXT
    );

create table apply(
    id INTEGER primary key AUTO_INCREMENT,
    orderId INTEGER,
    carId INTEGER,
    status BOOLEAN,
    createTime BIGINT
    );

create table logistics(
    orderId INTEGER primary key,
    info TEXT,
    createTime BIGINT
    );


alter table logistics CONVERT TO CHARACTER SET utf8 COLLATE utf8_general_ci;
